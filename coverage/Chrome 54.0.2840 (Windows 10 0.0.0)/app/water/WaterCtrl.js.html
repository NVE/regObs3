        };
 
        vm.openObservations = function () {
            var url = AppSettings.getObservationsUrl('Flood');
            $cordovaInAppBrowser.open(url, '_system' );
        };
 
        $scope.$on( '$ionicView.loaded', function(){} );
    }

        .module('RegObs')
        .controller('WaterCtrl', WaterCtrl);
